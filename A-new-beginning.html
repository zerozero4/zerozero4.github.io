std::map<student_t, mark_t> students = /* Intialize map here */
auto grade_o = students | where(is_grade_o); /* Returns a map containg students with grade O */
auto grade_o_students = grade_o | select_keys;  /* Returns a vector of student names */
auto sorted_students = grade_o_students | sort; /* Sorts the vector lexicographically */
